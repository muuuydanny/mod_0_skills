glass_cup

attributes:

current_volume (integer): 100

is_clean (boolean): true

material (string): "glass"

contents_of_cup (string): "water"

methods:

take_drink: takes 5 away from current_volume. I.e. current_volume => 95

fill: adds 5 to current_volume. I.e. current_volume => 100

spill_cup: sets current_volume to 0. I.e. current_volume => 0

use_cup: sets is_clean to false. I.e. is_clean => false
